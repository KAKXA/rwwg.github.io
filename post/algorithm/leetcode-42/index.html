class Solution {
    public int trap(int[] height) {
        int n = height.length;
        if (n == 0) {
            return 0;
        }

        int[] leftMax = new int[n];
        leftMax[0] = height[0];
        // leftMax[i]è®°å½•iå·¦ä¾§(åŒ…æ‹¬i)æœ€å¤§çš„é«˜åº¦
        for (int i = 1; i < n; ++i) {
            leftMax[i] = Math.max(leftMax[i - 1], height[i]);
        }

        int[] rightMax = new int[n];
        rightMax[n - 1] = height[n - 1];
        // rightMax[i]è®°å½•iå³ä¾§(åŒ…æ‹¬i)æœ€å¤§çš„é«˜åº¦
        for (int i = n - 2; i >= 0; --i) {
            rightMax[i] = Math.max(rightMax[i + 1], height[i]);
        }

        int ans = 0;
        // æ¯ä¸ªä½ç½®èƒ½è“„çš„æœ€å¤§å€¼æ˜¯ä¸¤è¾¹æœ€å¤§é«˜åº¦çš„æœ€å°å€¼-å½“å‰ä¸‹æ ‡çš„é«˜åº¦
        for (int i = 0; i < n; ++i) {
            ans += Math.min(leftMax[i], rightMax[i]) - height[i];
        }
        return ans;
    }
}

class Solution {
    public int trap(int[] height) {
        int ans = 0;
        int left = 0, right = height.length - 1;
        int leftMax = 0, rightMax = 0;
        // æ ¹æ®ä¸‹é¢çš„é€»è¾‘,å½“lefté‡åˆ°ç¬¬ä¸€ä¸ªæœ€é«˜æŸ±å­æ—¶,å®ƒä¸å¯èƒ½è¿‡å»,å› ä¸ºrightæŒ‡ä¸åˆ°é«˜äºæœ€é«˜æŸ±çš„ä½ç½®
        // æ‰€ä»¥leftå’Œrightç›¸ç¢°åœ¨iæ—¶,iä¸€å®šæ—¶æœ€é«˜æŸ±,åœ¨è¿™ä¸ªæŸ±ä¸å¯èƒ½è“„æ°´
        while (left < right) {
            leftMax = Math.max(leftMax, height[left]);
            rightMax = Math.max(rightMax, height[right]);
            // å·¦è¾¹maxæ¯”å³è¾¹maxå°,è¯´æ˜leftå¤„ä¸€å®šä¼šè¢«rightMaxå¤„boundä½
            // åä¹‹,è¯´æ˜rightä¸€å®šä¼šè¢«rightMaxå¤„boundä½
            // å¤„ç†å®Œä»¥å,left/rightå¤„çš„ç»“æœå·²ç»ç¡®å®šäº†,åªæ›´æ–°å…¶ä¸­ä¸€ä¸ªæŒ‡é’ˆ
            if (height[left] < height[right]) {

                ++left;
            } else {
                ans += rightMax - height[right];
                --right;
            }
        }
        return ans;
    }
