Leetcode 42. 接雨水

给定 n 个非负整数表示每个宽度为 1 的柱子的高度图，计算按此排列的柱子，下雨之后能接多少雨水。

class Solution {
    public int trap(int[] height) {
        int n = height.length;
        if (n == 0) {
            return 0;
        }

        int[] leftMax = new int[n];
        leftMax[0] = height[0];
        // leftMax[i]记录i左侧(包括i)最大的高度
        for (int i = 1; i < n; ++i) {
            leftMax[i] = Math.max(leftMax[i - 1], height[i]);
        }

        int[] rightMax = new int[n];
        rightMax[n - 1] = height[n - 1];
        // rightMax[i]记录i右侧(包括i)最大的高度
        for (int i = n - 2; i >= 0; --i) {
            rightMax[i] = Math.max(rightMax[i + 1], height[i]);
        }

        int ans = 0;
        // 每个位置能蓄的最大值是两边最大高度的最小值-当前下标的高度
        for (int i = 0; i < n; ++i) {
            ans += Math.min(leftMax[i], rightMax[i]) - height[i];
        }
        return ans;
    }
}

import java.util.Deque;
import java.util.LinkedList;

class Solution {
    public int trap(int[] height) {
        int ans = 0;
        Deque<Integer> stack = new LinkedList<Integer>();
        int n = height.length;
        // height 按照从高到低的顺序存储柱子的高度
        for (int i = 0; i < n; ++i) {
            // 当前的柱子i高度比栈顶下标top1的高,则i到top2之间必然由一个高度为min(h[i],h[top2]) - top1, 宽度为(i - top2 - 1)的矩形凹槽
            while (!stack.isEmpty() && height[i] > height[stack.peek()]) {
                // 取出矩形的底
                int top = stack.pop();
                // 如果空了,说明矩形左侧的边根本不存在,即水会从整个数组最左侧漏出去
                if (stack.isEmpty()) {
                    break;
                }
                // 取出矩形的左侧的边
                int left = stack.peek();
                // 得到宽度
                int currWidth = i - left - 1;
                // 得到高度 min(h[top2], h[i]) - top1
                int currHeight = Math.min(height[left], height[i]) - height[top];
                // 把矩形面积加到结果中
                ans += currWidth * currHeight;
            }
            // 在之前的循环中,会把比小的所有元素全部泵出
            // 所以直接把i压入栈,一定能保证它下面的元素比它大
            stack.push(i);
        }
        return ans;
    }
}

class Solution {
    public int trap(int[] height) {
        int ans = 0;
        int left = 0, right = height.length - 1;
        int leftMax = 0, rightMax = 0;
        // 根据下面的逻辑,当left遇到第一个最高柱子时,它不可能过去,因为right指不到高于最高柱的位置
        // 所以left和right相碰在i时,i一定时最高柱,在这个柱不可能蓄水
        while (left < right) {
            leftMax = Math.max(leftMax, height[left]);
            rightMax = Math.max(rightMax, height[right]);
            // 左边max比右边max小,说明left处一定会被rightMax处bound住
            // 反之,说明right一定会被rightMax处bound住
            // 处理完以后,left/right处的结果已经确定了,只更新其中一个指针
            if (height[left] < height[right]) {
                ans += leftMax - height[left];
                ++left;
            } else {
                ans += rightMax - height[right];
                --right;
            }
        }
        return ans;
    }
}
