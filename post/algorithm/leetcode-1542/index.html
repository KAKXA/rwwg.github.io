import java.util.Arrays;
import java.util.HashMap;
import java.util.Map;
// 前缀和的题
// 要么就添加一个-1的下标,这种比较符合直觉
// 要么开n+1的空间
// 这题我一开始做错的原因就是把0当作起始下标
// 导致以下标0为开头的字串被忽略了

// 键值到整数的映射很好找而且键值空间不大的情况下,不要使用hash表而是使用数组

class Solution {
    public int longestAwesome(String s) {
        // 由于在一个子串中出现至多一个计数为奇的数字就算一个回文串
        // 可以发现每个数字都只有两个状态,0和1
        // 因此状态压缩
        
        int[] left = new int[1 << 10];
        Arrays.fill(left, Integer.MAX_VALUE);
        left[0] = -1;

        int seq = 0, res = 1;
        for(int i = 0; i < s.length(); i++) {
            seq ^= 1 << (s.charAt(i) - '0');
            left[seq] = Math.min(left[seq], i);
            for(int o = 0; o < 10; ++o) {
                res = Math.max(res, i - left[seq ^ (1 << o)]);
            }
        }
        return seq == 0 ? s.length( ) : res;
    }
}
