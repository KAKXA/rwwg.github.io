# 将strs这个字符串列表转为前缀树
def f(trie, strs):
    for s in strs:
        d = trie
        for i in s:
            d = d.setdefault(i,{})
        # 用None表示结束标志
        d[None] = None
# 检查前缀树中有没有s这个字符串
def check(trie, s):
    d = trie
    for i in s:
        if i in d:
            d = d[i]
        else:
            return False
    return None in d

# 用法
strs = ['1234','1235','1345','1111','2345','1222','2223']
trie = dict()
f(trie, strs)
print(check(trie, '2223'))
