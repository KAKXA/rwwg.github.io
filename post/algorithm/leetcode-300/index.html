class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        n = len(nums)
        if not n:
            return 1
        d = [0] * (n + 1)
        d[1] = nums[0]
        le = 1
        for i in range(1, n):
            t = nums[i]
            if d[le] < t:
                d[le + 1] = t
                le += 1
            else:
                l = 1
                r = le + 1
                while l != r:
                    mid = (l + r) // 2
                    if d[mid] < t:
                        l = mid + 1
                    else:
                        r = mid
                d[l] = t
        return le
