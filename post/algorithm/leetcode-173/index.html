class BSTIterator:

    def __init__(self, root: TreeNode):
        self.cur = root
        self.arr = []

    def next(self) -> int:
        while self.cur:
            self.arr.append(self.cur)
            self.cur = self.cur.left

        cur = self.arr.pop()
        self.cur = cur.right
        return cur.val

    def hasNext(self) -> bool:
        return self.cur != None or len(self.arr) != 0
