class Solution:
    def dailyTemperatures(self, T: List[int]) -> List[int]:
        l = len(T)
        res= [0] * l
        for i in range(l - 2, -1, -1):
            now = i + 1
            while T[i] >= T[now]:
                if res[now]:
                    now += res[now]
                else:
                    break
            else:
                res[i] = now - i
        return res
