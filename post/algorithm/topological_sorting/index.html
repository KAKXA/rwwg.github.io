import java.util.*;

class Test {
    public int[] ts(ArrayList<List<Integer>> graph) {
        int cnt = 0;
        int n = graph.size();
        int[] res = new int[n];

        Queue<Integer> queue = new LinkedList<>();
        int[] indeg = new int[n];

        // 统计每个节点的入度
        for(List<Integer> le: graph) {
            for(int e: le) {
                ++indeg[e];
            }
        }
        // 入度为0的点优先加入队列
        for(int i = 0; i < n; ++i) {
            if (indeg[i] == 0) {
                queue.offer(i);
            }
        }

        while(!queue.isEmpty()) {
            int cur = queue.poll();
            // 把出队列的节点加到结果数组中
            res[cnt++] = cur;
            for (int end: graph.get(cur)) {
                --indeg[end];
                // 入度被置为0时,意味着在它前面形成约束的节点已经全部入队列了
                if (indeg[end] == 0) {
                    queue.offer(end);
                }
            }
        }
        if (cnt != n) {
            System.out.println("Fail to get all vertexes sorted.");
        }
        return res;
    }
    public static void main(String[] args) {
        // 测试用图
        Test s = new Test();
        ArrayList<List<Integer>> graph = new ArrayList<>();
        graph.add(Arrays.asList(1));
        graph.add(Arrays.asList(2, 3, 4));
        graph.add(Arrays.asList(4));
        graph.add(Arrays.asList());
        graph.add(Arrays.asList(3));

        int[] res = s.ts(graph);
        for(int i: res) {
            System.out.println(i);
        }
    }
}

模板题

207. 课程表

你这个学期必须选修 numCourses 门课程，记为 0 到 numCourses - 1 。

在选修某些课程之前需要一些先修课程。 先修课程按数组 prerequisites 给出，其中 prerequisites[i] = [ai, bi] ，表示如果要学习课程 ai 则 必须 先学习课程 bi 。

例如，先修课程对 [0, 1] 表示：想要学习课程 0 ，你需要先完成课程 1 。

请你判断是否可能完成所有课程的学习？如果可以，返回 true ；否则，返回 false 。

import java.util.ArrayList;
import java.util.LinkedList;
import java.util.List;
import java.util.Queue;

class Solution {
    // edges.get(i) 表示依赖于i的课程
    // 也就是完成课程i后,可以减去1入度的节点
    List<List<Integer>> edges;
    // 记录每个课程的入度
    int[] indeg;

    public boolean canFinish(int numCourses, int[][] prerequisites) {
        edges = new ArrayList<List<Integer>>();
        for (int i = 0; i < numCourses; ++i) {
            edges.add(new ArrayList<Integer>());
        }
        //初始化indeg和edges
        indeg = new int[numCourses];
        for (int[] info : prerequisites) {
            edges.get(info[1]).add(info[0]);
            ++indeg[info[0]];
        }

        // 队列,出队列的顺序也就是拓扑排序的顺序
        // 第一个出队列的节点一定是入度为0的节点
        Queue<Integer> queue = new LinkedList<Integer>();
        for (int i = 0; i < numCourses; ++i) {
            if (indeg[i] == 0) {
                queue.offer(i);
            }
        }

        int visited = 0;
        while (!queue.isEmpty()) {
            // 记录出队列的节点数
            ++visited;
            int u = queue.poll();
            
            // 拿到依赖于节点u的节点,它们可以解放啦!
            for (int v: edges.get(u)) {
                --indeg[v];
                if (indeg[v] == 0) {
                    // 完全解放了就可以放入队列了!
                    queue.offer(v);
                }
            }
        }
        return visited == numCourses;
    }
}
