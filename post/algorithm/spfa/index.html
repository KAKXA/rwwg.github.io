import java.util.*;

class Edge {
    int end;
    int weight;
    public Edge(int end, int weight) {
        this.end = end;
        this.weight = weight;
    }
}
class Test {
    public int[] spfa(int src, ArrayList<List<Edge>> graph, int irreachable_tag) {
        int n = graph.size();
        Queue<Integer> q = new LinkedList<Integer>();
        int[] dist = new int[n];
        boolean[] visited = new boolean[n];
        // åˆ°è¾¾è¯¥ç‚¹çš„æœ€çŸ­è·¯å¾„ç»è¿‡çš„èŠ‚ç‚¹æ•°
        int[] count = new int[n];
        for(int i = 0; i < n; ++i) {
            dist[i] = irreachable_tag;
            visited[i] = false;
        }
        dist[src] = 0;
        visited[src] = true;
        q.offer(src);

        while (!q.isEmpty()) {
            int cur = q.poll();
            visited[cur] = false;
            System.out.println("Polled from queue: " + Integer.toString(cur));
            // éå†å’Œcurç›´æ¥ç›¸è¿çš„æ‰€æœ‰ç‚¹,æ›´æ–°å…¶åˆ°srcçš„æœ€çŸ­è·ç¦»

                if (dist[node.end] != irreachable_tag) {
                    if (dist[cur] + node.weight < dist[node.end]) {
                        dist[node.end] = dist[cur] + node.weight;
                        count[node.end] = count[cur] + 1;
                        if (count[node.end] > n) {
                            System.out.println("There is at least one negative cycle in the graph.");
                            Arrays.fill(dist, irreachable_tag);
                            return dist;
                        }
                        if (!visited[node.end]) {
                            visited[node.end] = true;
                            q.offer(node.end);
                        }
                    }
                } else {
                    dist[node.end] = dist[cur] + node.weight;
                    if (!visited[node.end]) {
                        visited[node.end] = true;
                        q.offer(node.end);
                    }
                }
            }
        }
        return dist;
    }
