import java.util.HashMap;
import java.util.Map;

class FSM {
    public String state = "start";
    public int sign = 1;
    public long res = 0;
    public Map<String, String[]> transitions = new HashMap<>();
    public FSM() {
        transitions.put("start", new String[] {"start", "sign", "number", "finish"});
        transitions.put("sign", new String[] {"finish", "finish", "number", "finish"});
        transitions.put("number", new String[] {"finish", "finish", "number", "finish"});
        transitions.put("finish", new String[] {"finish", "finish", "finish", "finish"});
    }

    public int get_col(char c) {
        // é€šè¿‡å­—ç¬¦è·å–å…¶å±äºæ•°ç»„ä¸­å“ªä¸€åˆ—çš„æƒ…å†µ
        if (c == ' ') {
            return 0;
        } else if (c == '+' || c == '-') {
            return 1;
        } else if (Character.isDigit(c)) {
            return 2;
        } else {
            return 3;
        }
    }

    public void get(char c) {
        // æ ¹æ®å­—ç¬¦cè¿›è¡ŒçŠ¶æ€å˜åŒ–
        state = transitions.get(state)[get_col(c)];
        if (state.equals("number")) {
            res = res * 10 + c - '0';
            res = sign == 1 ? Math.min(Integer.MAX_VALUE, res) : Math.min(-(long)Integer.MIN_VALUE, res);
        } else if (state.equals("sign")) {
            if (c == '+') {
                sign = 1;
            } else {
                sign = -1;
            }
        }
    }
}


class Solution {
    public int myAtoi(String str) {
        FSM fsm = new FSM();
        for(char c: str.toCharArray()) {
            fsm.get(c);
        }
        return (int)(fsm.sign * fsm.res);
    }
}
