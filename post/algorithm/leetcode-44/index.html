class Solution:
    def isMatch(self, s: str, p: str) -> bool:
        ls = len(s)
        ps = len(p)
	# dp[i][j]表示模式串前j个字符能否匹配字符串前i个字符
        dp = [[False] * (ps + 1) for _ in range(ls + 1)]
	# 空模式串可以匹配空字符串
        dp[-1][-1] = True
	# (i + 1)长的模式串能否匹配空字符串?唯一能的可能性是它全为'*'
        for i in range(ps):
            dp[-1][i] = (p[:i + 1] == '*' * (i + 1))
	# 那(i + 1)长的字符串能否被空模式串匹配呢?当然是不能.
        for i in range(ls):
            for j in range(ps):
		# 转移方程
                if s[i] == p[j] or p[j] == '?':
                    dp[i][j] = dp[i - 1][j - 1]
                elif p[j] == '*':
                    dp[i][j] = dp[i - 1][j] or dp[i][j - 1]
        return dp[ls - 1][ps - 1]
