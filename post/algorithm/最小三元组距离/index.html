def f(a, b, c):
    def cal(a, b, c):
        return abs(a - b) + abs(a - c) + abs(b - c)
    l, m, n = len(a), len(b), len(c)
    res = float('inf')
    i, j, k = 0, 0, 0
    while i < l and j < m and k < n:
        res = min(res, cal(a[i], b[j], c[k]))
        if a[i] <= b[j] and a[i] <= c[k]:
            i += 1
        elif b[j] <= a[i] and b[j] <= c[k]:
            j += 1
        else:
            k += 1
    return res
