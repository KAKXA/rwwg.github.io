class Solution:
    def maximalRectangle(self, matrix: List[List[str]]) -> int:
        m = len(matrix)
        if not m:
            return 0
        n = len(matrix[0])
        if not n:
            return 0

        max_area = 0
        for i in matrix:
            for j in range(len(i)): i[j] = False if i[j] == '0' else True
                
        if matrix[0][0]:
            matrix[0][0] = {(0,0,1)}
            max_area = max(max_area, 1)
        for i in range(1, m):
            if matrix[i][0]:
                s = set()
                if matrix[i - 1][0]:
                    for pair in matrix[i - 1][0]:
                        s.add((pair[0],pair[1],pair[2] + 1))
                        max_area = max(max_area, pair[2] + 1)
                s.add((i, 0, 1))
                max_area = max(max_area, 1)
                matrix[i][0] = s

        for i in range(1, n):
            if matrix[0][i]:
                s = set()
                if matrix[0][i - 1]:
                    for pair in matrix[0][i - 1]:
                        s.add((pair[0],pair[1],pair[2] + 1))
                        max_area = max(max_area, pair[2] + 1)
                s.add((0, i, 1))
                max_area = max(max_area, 1)
                matrix[0][i] = s

        for i in range(1, m):
            for j in range(1, n):
                if matrix[i][j]:
                    s = set()
                    if matrix[i - 1][j] and matrix[i][j - 1]:
                        for pair_1 in matrix[i - 1][j]:
                            for pair_2 in matrix[i][j - 1]:
                                if pair_1[:2] == pair_2[:2]:
                                    area = (i - pair_1[0] + 1) * (j - pair_1[1] + 1)
                                    s.add((pair_1[0], pair_1[1], area))
                                    max_area = max(max_area, area)

                    if matrix[i - 1][j]:
                        for pair in matrix[i - 1][j]:
                            if pair[1] == j:
                                s.add((pair[0],pair[1],pair[2] + 1))
                                max_area = max(max_area, pair[2] + 1)

                    if matrix[i][j - 1]:
                        for pair in matrix[i][j - 1]:
                            if pair[0] == i:
                                s.add((pair[0],pair[1],pair[2] + 1))
                                max_area = max(max_area, pair[2] + 1)
                    s.add((i, j, 1))
                    max_area = max(max_area, 1)
                    matrix[i][j] = s
        return max_area
