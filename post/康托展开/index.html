定义

康托展开(Cantor Expasion)提供了n阶对称群$S_n$到定义在$[0,n!-1]$的全体正整数集的一个群$G$的同构.通俗来说,$s\in S_n$在$G$中的像是其在n阶置换中的字典序(从0开始).

from math import factorial
def cantor(perm):
    n = len(perm)
    res = 0
    rem = [i for i in range(n)]
    for i in range(n):
        res += rem[perm[i]] * factorial(n - i - 1)
        for j in range(perm[i] + 1, n):
            rem[j] -= 1
    return res

from math import factorial
def reversed_cantor(num, n):
    res = []
    rem = [i for i in range(n)]
    for i in range(n):
        t = int(num / factorial(n - i - 1))
        res.append(rem[t])
        for j in range(t, n - 1):
            rem[j] = rem[j + 1]
        num %= factorial(n - i - 1)
    return res
